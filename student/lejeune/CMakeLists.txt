CMAKE_MINIMUM_REQUIRED(VERSION 3.1)
PROJECT(Lejeune CXX)

# compiler settings
IF(CMAKE_CXX_COMPILER_ID MATCHES "MSVC")
    ADD_DEFINITIONS(-D_CRT_SECURE_NO_WARNINGS -D_SCL_SECURE_NO_DEPRECATE -D_USE_MATH_DEFINES)
    SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /MP")  # parallel build with MSVC
ENDIF()

ADD_EXECUTABLE(close close.cpp)
ADD_EXECUTABLE(closeb closeb.cpp)
ADD_EXECUTABLE(open open.cpp)
ADD_EXECUTABLE(openb openb.cpp)
